module sub_task;
    reg [7:0] a,b;
    reg [7:0] result;

task sub_8bit
    input [7:0] a, b;
    output [7:0] diff;

begin
    diff = a-b;
    $display("diff = %d",diff);
end
endtask

initial begin
     a = 8'd15;
     b = 8'd40;
     sub_8bit(a, b, result);
     $finish;
    end

endmodule
