task reverse_bits;
        input [7:0] in_data;      // Input 8-bit data
        output [7:0] out_data;    // Output 8-bit reversed data
        integer j;
    begin
        out_data = 8'b0;          // Initialize output to zero
        for (j = 0; j < 8; j = j + 1) begin
            out_data[j] = in_data[7 - j];  // Reverse the bit order
        end
        $display("Input = %b, Reversed = %b", in_data, out_data);
    end
    endtask
