module double_add_module;
   integer a,b,result;
//Task 1
task add_task;
        input x, y;
        output sum;
        begin
            sum = x + y;
        end
    endtask

// Task 2: 
task double_add_task;
        input  p, q;
        output total;
        integer temp;
        begin
            add_task(p, q, temp);
            add_task(temp, q, total);
            $display("Double Add Result = %0d", total);
        end

    endtask

 initial begin
        a = 5;
        b = 3;
        double_add_task(a, b, result);   

        a = 10;
        b = 4;
        double_add_task(a, b, result);

        $finish;
    end

endmodule
