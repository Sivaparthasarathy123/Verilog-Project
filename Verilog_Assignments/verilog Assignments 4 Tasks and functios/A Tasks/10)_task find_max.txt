module find_max;
      reg [3:0] a, b, c, d;
      reg [3:0] max_val;

task find_max;
        input [3:0] a, b, c, d;
        output [3:0] max;
        begin
            max = a;
            if (b > max) max = b;
            if (c > max) max = c;
            if (d > max) max = d;
            $display("Maximum = %0d", max);
        end
endtask
initial begin
    a = 4'd12; b = 4'd4; c = 4'd6; d = 4'd11;
    find_max(a, b, c, d, max_val);
    $finish;
    end

endmodule
