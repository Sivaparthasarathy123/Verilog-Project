//usage of $monitoron and $monitoroff to control monitoring.

module monitoron_example;
    reg clk;
    
    initial clk = 0;
    always #5 clk = ~clk;

    initial begin
        $monitoron;
        $monitor("Time=%0t clk=%b", $time, clk);
        #15 $monitoroff;
        #30 $finish;
    end
endmodule
