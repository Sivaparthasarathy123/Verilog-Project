//Demonstrate $monitor to track signal changes in a clock and data signal.

module track_signal;
    reg clk, data;

    initial begin
       clk = 0;
       data = 0;
    end 
    always clk = ~clk;
    always data = ~data;

initial begin
       $monitor("Time=%0t | clk = %b | data = %b", $time, clk, data);
end
endmodule
