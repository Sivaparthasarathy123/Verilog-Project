module func_count_ones;
    function [7:0] count1;
        input [7:0] x;
        integer i;
        begin
            count1 = 0;
            for(i=0;i<8;i++)
                if(x[i]) 
                 count1 = count1 + 1;
        end
    endfunction

    initial $display("Ones = %0d", count1(8'b10110110));
