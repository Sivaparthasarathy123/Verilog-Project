//a testbench that uses $display and $monitor together.

module disp_mon_tb;
    reg a, b;
    wire y;

    assign y = a & b;

    initial begin
        $display("Time=%0t Start Simulation", $time);
        $monitor("Time=%0t | a=%b | b=%b | y=%b", $time, a, b, y);
        a = 0; b = 0; #10;
        a = 1; b = 0; #10;
        a = 1; b = 1; #10;
        $finish;
    end
endmodule
